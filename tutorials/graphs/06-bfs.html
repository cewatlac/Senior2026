What is BFS?

Breadth-First Search explores a graph layer by layer — all nodes at distance 1 before distance 2, and so on. It uses a queue (FIFO) to maintain this order.
If all edges have equal cost, BFS gives the shortest path in number of edges from the source to every reachable node.

Shortest path in unweighted graphs
Minimum moves / minimum steps
Nearest source (multi-source BFS)
Layer-by-layer expansion
Not for arbitrary weighted graphs → use Dijkstra

The Queue is the Whole Idea

The 4 BFS operations in every template:

  1. Push the source into the queue, set dist = 0
  2. Pop the front node u
  3. For each unvisited neighbor v: set dist[v] = dist[u]+1, push v
  4. Repeat until queue is empty
FIFO Queue — Why BFS stays in order

Nodes are always processed in non-decreasing distance order — that's the invariant.

The BFS Invariant

When u is first discovered from v:    dist[u] = dist[v] + 1

This is the shortest possible distance in any unweighted graph. BFS never revisits, so the first time is always optimal.

Why it's correct

BFS processes nodes in non-decreasing distance order. So when we first reach node u at distance d, no shorter path can exist — all nodes at distance d-1 have already been fully processed.

The C++ check

if(dist[v] == -1){          // unvisited?
    dist[v] = dist[u] + 1;  // optimal!
    q.push(v);
}

dist[v] == -1 is the visited check. Never use a separate vis[] array if you already have dist[].

Complexity

O(V+E)Time
O(V)Space (queue)

Every vertex is enqueued and dequeued exactly once → O(V)
Every edge is examined at most twice (once per endpoint) → O(E)
Total: O(V + E) for adjacency list.

Adjacency Matrix → O(V²) — avoid on large sparse graphs
Adjacency List → O(V+E) — default for competitive programming

Template 1 — Layer BFS (Count Steps)

Use when the answer is number of BFS layers — minutes/rounds/waves of simultaneous spreading. Process the queue by snapshot size: sz = q.size() before the inner loop.

// Template 1: Layer BFS — returns distance to farthest reachable node
int bfs_layers(const vector<vector<int>>& adj, int src){
    int n = (int)adj.size();
    vector<int> vis(n, 0);
    queue<int> q;
    q.push(src);
    vis[src] = 1;
    int steps = 0;

    while(!q.empty()){
        int sz = (int)q.size();   // snapshot of current layer
        while(sz--){
            int u = q.front(); q.pop();
            for(int v : adj[u]){
                if(!vis[v]){
                    vis[v] = 1;
                    q.push(v);
                }
            }
        }
        steps++;
    }
    return steps - 1; // distance to farthest layer
}

Template 2 — BFS with Path Reconstruction

Uses a vector<tuple<State,int,Label>> as both queue and node store. Each entry holds the current state, the index of its parent, and the edge label used to reach it. A recursive print function walks parents to reconstruct the path. Replace State, Label, and the neighbor loop with your problem's specifics.

// Template 2: BFS with Path Reconstruction (generic)
// State  — type of each BFS node (int, pair, etc.)
// Label  — type of edge label to print on the path (char, int, string, etc.)

void print(vector<tuple<State,int,Label>>& q, int idx){
    auto [cur, par, lbl] = q[idx];
    if(idx) print(q, par);   // recurse to root first
    cout << lbl;              // then print label on the way back
}

void solve(State start, /* problem params */){
    unordered_set<State> vis;
    vector<tuple<State,int,Label>> q;

    q.emplace_back(start, -1, Label{} /* dummy label for root */);
    vis.insert(start);

    for(int i = 0; i < (int)q.size(); i++){
        auto [cur, par, lbl] = q[i];

        for(auto& [nxt, edge] : getNeighbors(cur)){  // your neighbor generator
            if(vis.insert(nxt).second){
                q.emplace_back(nxt, i, edge);
                if(isGoal(nxt)){          // stop early if target found
                    i = (int)q.size();    // break outer loop too
                    break;
                }
            }
        }
    }
    print(q, (int)q.size() - 1);
    cout << '\n';
}
Reconstructs full path without extra parent array
All visited nodes stay in q (inspectable)
Works for any state type — int, pair, string, …
Stop-early: set i = q.size() to break both loops

Template 3 — Layer BFS (steps counter, no dist[])

Processes the queue layer by layer. steps++ fires before the inner loop so that returning inside the loop gives the correct distance with no off-by-one. Returns 0 if src == goal, -1 if unreachable.

// Template 3: Layer BFS — returns steps to reach goal, -1 if unreachable
int bfs_steps(const vector<vector<int>>& adj, int src, int goal){
    if(src == goal) return 0;             // already there
    int n = (int)adj.size();
    vector<bool> vis(n, false);
    queue<int> q;

    vis[src] = true;
    q.push(src);
    int steps = 0;

    while(!q.empty()){
        int sz = (int)q.size();   // snapshot of this layer
        steps++;                          // about to process layer #steps
        while(sz--){
            int u = q.front(); q.pop();
            for(int v : adj[u]){
                if(!vis[v]){
                    if(v == goal) return steps;  // early exit — no off-by-one
                    vis[v] = true;
                    q.push(v);
                }
            }
        }
    }
    return -1;  // goal unreachable
}
Spreading fire / virus / waves
Minimum rounds until completion
Layer-end logic: insert code after inner while(sz--)
No dist[] — use Template 4 if you need per-node distances

Template 4 — Standard BFS with dist[] for All Nodes

The all-purpose BFS template. Returns distances from one source to all reachable nodes. Use this by default.

// Template 4: Standard BFS — dist[] for all nodes (most common)
#include <bits/stdc++.h>
using namespace std;

vector<int> bfs_all_dist(const vector<vector<int>>& adj, int src){
    int n = (int)adj.size();
    vector<int> dist(n, -1);
    queue<int> q;

    dist[src] = 0;
    q.push(src);

    while(!q.empty()){
        int u = q.front();
        q.pop();

        for(int v : adj[u]){
            if(dist[v] == -1){      // unvisited
                dist[v] = dist[u] + 1;
                q.push(v);
            }
        }
    }
    return dist; // dist[i] = -1 means unreachable
}

0-1 BFS (Deque)

When edge weights are only 0 or 1, use a deque: weight 0 → push_front (free move), weight 1 → push_back (paid move). This keeps nodes in non-decreasing distance order and runs in O(V+E).

// 0-1 BFS — deque trick for {0,1} edge weights
const int INF = 1e9;

vector<int> bfs01(const vector<vector<pair<int,int>>>& adj, int src){
    int n = (int)adj.size();
    vector<int> dist(n, INF);
    deque<int> q;

    dist[src] = 0;
    q.push_front(src);

    while(!q.empty()){
        int u = q.front(); q.pop_front();
        for(auto [v, w] : adj[u]){
            if(dist[u] + w < dist[v]){
                dist[v] = dist[u] + w;
                if(w == 0) q.push_front(v);  // free → front
                else        q.push_back(v);   // cost → back
            }
        }
    }
    return dist;
}
Edge cost ∈ {0,1}
Arbitrary weights → Dijkstra
All weights equal → standard BFS

BFS on Disconnected Graphs

A single BFS call only visits one connected component. Loop over all nodes and start a BFS from each unvisited node to cover the whole graph.

// Disconnected graph: BFS from every unvisited source
vector<int> bfs_all_components(const vector<vector<int>>& adj){
    int n = (int)adj.size();
    vector<int> vis(n, 0), order;
    queue<int> q;

    for(int s = 0; s < n; s++){
        if(vis[s]) continue;       // already visited
        vis[s] = 1;
        q.push(s);

        while(!q.empty()){
            int u = q.front(); q.pop();
            order.push_back(u);
            for(int v : adj[u]){
                if(!vis[v]){
                    vis[v] = 1;
                    q.push(v);
                }
            }
        }
    }
    return order;
}

BFS on Implicit Graphs

Many BFS problems don't have an explicit adjacency list. The "node" is a state, and neighbors are generated on the fly.

8-puzzleboard encodingswap 0 with adjacent tile
"Getting Zero"value mod 32768×2 or +1 transitions

Rule: If the problem has extra state beyond position, make that extra state part of the BFS node: state = (pos, extra). 
// Implicit graph BFS — generic pattern
map<State, int> dist;
queue<State> q;
State start = ...;
dist[start] = 0;
q.push(start);

while(!q.empty()){
    State cur = q.front(); q.pop();
    for(State nxt : neighbors(cur)){   // generate on the fly
        if(!dist.count(nxt)){
            dist[nxt] = dist[cur] + 1;
            q.push(nxt);
        }
    }
}

Kahn's Algorithm — BFS Topological Sort

Repeatedly remove nodes with indegree 0 using a queue. When a neighbor's indegree drops to 0, it joins the queue.

// Kahn's BFS topological sort
vector<int> kahn_topo(const vector<vector<int>>& adj){
    int n = (int)adj.size();
    vector<int> indeg(n, 0);
    for(int u = 0; u < n; u++)
        for(int v : adj[u]) indeg[v]++;

    queue<int> q;
    for(int i = 0; i < n; i++)
        if(indeg[i] == 0) q.push(i);

    vector<int> order;
    while(!q.empty()){
        int u = q.front(); q.pop();
        order.push_back(u);
        for(int v : adj[u])
            if(--indeg[v] == 0) q.push(v);
    }

    if((int)order.size() != n) return {}; // cycle detected
    return order;
}
Detects cycles (output size < n → cycle exists)
UVA 11686 uses exactly this pattern

Bipartite Check with BFS

Color nodes with 2 colors, alternating per BFS level. If a neighbor has the same color → not bipartite.

bool is_bipartite(const vector<vector<int>>& adj){
    int n = (int)adj.size();
    vector<int> color(n, -1);
    queue<int> q;

    for(int s = 0; s < n; s++){
        if(color[s] != -1) continue;
        color[s] = 0;
        q.push(s);

        while(!q.empty()){
            int u = q.front(); q.pop();
            for(int v : adj[u]){
                if(color[v] == -1){
                    color[v] = color[u] ^ 1; // flip color
                    q.push(v);
                }else if(color[v] == color[u]){
                    return false;            // same color → not bipartite
                }
            }
        }
    }
    return true;
}

Cycle Detection in Undirected Graph (BFS)

Track (node, parent) in the queue. If you see a visited neighbor that is not the parent → cycle found.

bool has_cycle_bfs(const vector<vector<int>>& adj){
    int n = (int)adj.size();
    vector<int> vis(n, 0);

    for(int s = 0; s < n; s++){
        if(vis[s]) continue;
        queue<pair<int,int>> q;
        q.push({s, -1});
        vis[s] = 1;

        while(!q.empty()){
            auto [u, p] = q.front(); q.pop();
            for(int v : adj[u]){
                if(!vis[v]){
                    vis[v] = 1;
                    q.push({v, u});
                }else if(v != p){
                    return true; // back edge → cycle
                }
            }
        }
    }
    return false;
}

Problem 5 — SPOJ BITMAP

Abstract: We want the distance from every cell to the nearest '1'. Running BFS separately from each cell is too expensive, so we reverse the viewpoint: start BFS from all '1' cells together. Then the first time we reach a cell is its minimum distance to any source.

  1. Read the grid and create dist initialized to -1.
  2. Push every '1' cell into the queue with distance 0.
  3. Pop a cell, try the 4 directions using di and dj.
  4. When an unvisited neighbor is valid, set its distance and push it.
  5. When BFS ends, every cell stores the nearest-source distance.

For each cell, compute distance to nearest '1'. Key insight: push all '1' cells into the queue at distance 0 simultaneously → multi-source BFS.

#include <bits/stdc++.h>
using namespace std;
int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int tc; cin >> tc;
    while(tc--){
        int n, m; cin >> n >> m;
        vector<string> grid(n);
        for(auto& r : grid) cin >> r;

        vector<vector<int>> dist(n, vector<int>(m, -1));
        queue<pair<int,int>> q;

        // Push ALL '1' cells first
        for(int i=0;i<n;i++) for(int j=0;j<m;j++)
            if(grid[i][j]=='1'){ dist[i][j]=0; q.push({i,j}); }

        int di[]={0,0,-1,1}, dj[]={-1,1,0,0};
        while(!q.empty()){
            auto [i,j] = q.front(); q.pop();
            for(int k=0;k<4;k++){
                int ni=i+di[k], nj=j+dj[k];
                if(ni<0||ni>=n||nj<0||nj>=m) continue;
                if(dist[ni][nj]!=-1) continue;
                dist[ni][nj]=dist[i][j]+1; q.push({ni,nj});
            }
        }
        for(int i=0;i<n;i++){ for(int j=0;j<m;j++) cout<<dist[i][j]<<' '; cout<<'\n'; }
    }
}

Problem 7 — SPOJ ONEZERO

Abstract: We do not build huge numbers directly. Instead, each state is only the remainder modulo n. From a remainder r, appending digit 0 or 1 creates two new remainders. BFS guarantees that the first time we reach remainder 0, we have the shortest answer in number of digits, and trying 0 before 1 keeps it lexicographically small within that length.

  1. Start from remainder 1 % n.
  2. For each popped state, try appending digits 0 and 1.
  3. Store parent information so the final string can be reconstructed.
  4. Stop when remainder 0 appears.
  5. Rebuild the answer by walking through parents backward.

Find the smallest multiple of n consisting only of digits 0 and 1. Model: state = remainder mod n, transitions = append digit 0 or 1, start at remainder (1%n), goal = remainder 0. Reconstruct digits via parent array.

#include <bits/stdc++.h>
using namespace std;

void print(vector<tuple<int,int,char>>& q, int idx){
    auto [cur, par, dg] = q[idx];
    if(idx) print(q, par);
    cout << dg;
}

void solve(int n){
    if(n==1){ cout << "1\n"; return; }
    vector<tuple<int,int,char>> q;
    q.emplace_back(1, -1, '1');
    vector<bool> vis(n);
    vis[1] = 1;
    for(int i = 0; i < (int)q.size(); i++){
        auto [cur, par, dg] = q[i];
        for(int d = 0; d < 2; d++){
            int nxt = (cur*10+d)%n;
            if(vis[nxt]) continue;
            vis[nxt] = 1;
            q.emplace_back(nxt, i, '0'+d);
            if(!nxt){ i = (int)q.size(); break; }
        }
    }
    print(q, (int)q.size()-1);
    cout << '\n';
}

int main(){
    int tc; cin >> tc;
    while(tc--){ int n; cin >> n; solve(n); }
}

Problem 4 — UVA 11686 (Kahn BFS Topological Order)

Abstract: A topological order exists only in a directed acyclic graph. Kahn’s algorithm repeatedly takes any node whose indegree is zero, places it in the answer, and removes its outgoing edges. If we process all nodes, the graph is acyclic; otherwise, a cycle blocked some nodes forever.

  1. Build the directed graph and compute indegrees.
  2. Push every zero-indegree node into the queue.
  3. Pop one node, append it to the order.
  4. Decrease indegrees of all outgoing neighbors.
  5. If a neighbor becomes zero, push it into the queue.
int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n, m;
    while(cin >> n >> m){
        if(n==0 && m==0) break;
        vector<vector<int>> adj(n);
        vector<int> indeg(n, 0);
        for(int i=0;i<m;i++){
            int u,v; cin>>u>>v; --u;--v;
            adj[u].push_back(v); indeg[v]++;
        }
        queue<int> q;
        for(int i=0;i<n;i++) if(indeg[i]==0) q.push(i);
        vector<int> order;
        while(!q.empty()){
            int u=q.front();q.pop();
            order.push_back(u);
            for(int v:adj[u]) if(--indeg[v]==0) q.push(v);
        }
        if((int)order.size()!=n) cout<<"IMPOSSIBLE\n";
        else for(int x:order) cout<<(x+1)<<"\n";
    }
}

Problem 8 — 8-Puzzle (Aizu ALDS1_13_B)

Abstract: The puzzle state is encoded as a 9-digit integer. Each move swaps the blank (0) with one of its four neighbors. BFS finds the minimum number of moves from start to goal.

  1. Encode the 3×3 board as an integer (each cell one decimal digit).
  2. BFS layer by layer from the start encoding to the goal encoding.
  3. Neighbors: decode, find 0, swap with each valid adjacent cell, re-encode.
  4. The layer count when goal is first dequeued is the answer.

Link: vjudge.net/problem/Aizu-ALDS1_13_B

#include <bits/stdc++.h>
using namespace std;

int encode(const vector<vector<int>>& b){
    int res = 0;
    for(auto& row: b) for(auto cell: row) res = res*10+cell;
    return res;
}
vector<vector<int>> decode(int code){
    vector<vector<int>> res(3, vector<int>(3));
    for(int i=2;~i;i--) for(int j=2;~j;j--){ res[i][j]=code%10; code/=10; }
    return res;
}
pair<int,int> findZero(const vector<vector<int>>& b){
    for(int i=0;i<3;i++) for(int j=0;j<3;j++) if(!b[i][j]) return {i,j};
    return {-1,-1};
}
vector<int> getNeig(int code){
    auto b = decode(code);
    vector<int> res;
    auto [i,j] = findZero(b);
    for(int di=0,dj=-1,k=0; k<4; k++){
        int ni=i+di, nj=j+dj;
        if(ni>=0&&ni<3&&nj>=0&&nj<3){ swap(b[i][j],b[ni][nj]); res.push_back(encode(b)); swap(b[i][j],b[ni][nj]); }
        swap(di,dj); if(k==1) dj=1;
    }
    return res;
}
int main(){
    vector<vector<int>> goal={{1,2,3},{4,5,6},{7,8,0}}, start(3,vector<int>(3));
    for(auto& r: start) for(auto& c: r) cin>>c;
    int startCode=encode(start), goalCode=encode(goal);
    if(startCode==goalCode){ cout<<0<<endl; return 0; }
    queue<int> q; q.push(startCode);
    unordered_set<int> vis={startCode};
    int steps=0;
    while(q.size()){
        int sz=q.size(); ++steps;
        while(sz--){
            int u=q.front(); q.pop();
            for(auto v: getNeig(u))
                if(vis.insert(v).second)
                    if(v==goalCode){ cout<<steps<<endl; return 0; }
                    else q.push(v);
        }
    }
}

Problem 9 — 8-Puzzle Multi-Query (LightOJ 1139)

Abstract: With many test cases all asking for distance to the same goal, it is more efficient to BFS backward from the goal once and store all reachable distances, then answer each query in O(1).

  1. Pre-compute BFS from the goal state, storing dist[state] for every reachable board.
  2. For each query, encode the start board and look up its distance.
  3. If the start state was never reached, output impossible.

Link: vjudge.net/problem/LightOJ-1139

#include <bits/stdc++.h>
using namespace std;
// encode / decode / findZero / getNeig — same as Problem 8 above

int main(){
    vector<vector<int>> goal={{1,2,3},{4,5,6},{7,8,0}};
    int goalCode = encode(goal);

    // Pre-BFS from goal
    queue<int> q; q.push(goalCode);
    unordered_map<int,int> vis = {{goalCode, 0}};
    int steps = 0;
    while(q.size()){
        int sz = q.size(); ++steps;
        while(sz--){
            int u = q.front(); q.pop();
            for(auto v: getNeig(u))
                if(vis.emplace(v, steps).second)
                    q.push(v);
        }
    }

    int t, tc=0; cin >> t;
    vector<vector<int>> start(3, vector<int>(3));
    while(t--){
        for(auto& r: start) for(auto& c: r) cin>>c;
        int startCode = encode(start);
        auto it = vis.find(startCode);
        cout << "Case " << ++tc << ": ";
        if(it==vis.end()) cout << "impossible\n";
        else              cout << it->second << '\n';
    }
}

Problem 10 — 8-Puzzle with Path Output (UVA 10085)

Abstract: Same BFS on 8-puzzle states, but instead of counting moves the program must output the move sequence (L/U/R/D) and the final sorted board. Uses Template 2's path-reconstruction pattern: the BFS queue is a vector<tuple> storing each state's parent index and the move label used to reach it.

  1. BFS runs until the queue is exhausted (goal = lexicographically maximum reachable state).
  2. The last entry in the queue is the goal; walk parents backward to reconstruct moves.
  3. Output the final board followed by the move string.

Link: vjudge.net/problem/UVA-10085

#include <bits/stdc++.h>
using namespace std;
// encode / decode / findZero same as Problem 8

vector<pair<int,char>> getMoves(int cur){
    vector<pair<int,char>> ret;
    auto grid = decode(cur);
    int i, j, di=0, dj=-1;
    findZero(grid, i, j);
    string s = "LURD";
    for(int k=0; k<4; k++){
        int ni=i+di, nj=j+dj;
        if(ni>=0&&ni<3&&nj>=0&&nj<3){
            swap(grid[i][j], grid[ni][nj]);
            ret.emplace_back(encode(grid), s[k]);
            swap(grid[i][j], grid[ni][nj]);
        }
        swap(di, dj); if(k==1) dj=1;
    }
    return ret;
}
void print(vector<tuple<int,int,char>>& q, int idx){
    if(idx){ auto [cur,par,move]=q[idx]; print(q,par); cout<<move; }
}
void solve(){
    unordered_set<int> vis;
    vector<vector<int>> grid(3,vector<int>(3));

BFS vs DFS — When to Use Which

Use BFS when:
Shortest path in unweighted graph
Minimum moves / steps
Minimum rounds / minutes
Nearest source (multi-source)
Layer-by-layer expansion
Bipartite check
Use DFS when:
Connected components
Cycle structure (directed)
Tree DP
Topological sort by DFS
Bridges / Articulation / SCC
Backtracking / exhaustive search
⚠ DFS does NOT guarantee shortest paths. Use BFS for minimum steps in unweighted graphs — DFS can return a longer path.

Common BFS Mistakes

Mistake 1 — Forgetting the visited check causes TLE / infinite loop

Without checking dist[v] == -1, the same node can be enqueued O(E) times, exploding the time complexity.

// WRONG — no visited check:
for(int v : adj[u]) q.push(v);

// CORRECT:
for(int v : adj[u]){
    if(dist[v] == -1){ dist[v] = dist[u]+1; q.push(v); }
}
Mistake 2 — Wrong C++ queue pop compile error or silent bug

q.pop() returns void in C++. Always use q.front() first, then q.pop().

// WRONG:
int u = q.pop(); // pop() returns void!

// CORRECT:
int u = q.front();
q.pop();
Mistake 3 — Using BFS on weighted graphs wrong answer

Standard BFS only works when all edge weights are equal.

Equal weights → BFS
{0,1} weights → 0-1 BFS (deque)
Arbitrary ≥ 0 weights → Dijkstra
Mistake 4 — Global distance++ on every pop wrong distances

Incrementing a global counter on every pop() is wrong — distance must come from dist[u] + 1, or from layer loops.

// WRONG:
int d = 0;
while(!q.empty()){ int u=q.front();q.pop(); d++; ... }

// CORRECT: use dist[] array
dist[v] = dist[u] + 1;
Mistake 5 — Multi-source: pushing sources one by one with step increments wrong distances

All sources must be pushed before BFS starts, all at distance 0. Not one at a time.

// WRONG: each source starts a separate BFS
for(auto s : sources) bfs(s); // distances overlap!

// CORRECT: push all sources at distance 0 first
for(auto [i,j] : ones){ dist[i][j]=0; q.push({i,j}); }
// then run ONE BFS

Full BFS Toolkit

All templates in one place for contest use.

#include <bits/stdc++.h>
using namespace std;

// 1) Standard BFS distances
vector<int> bfs_dist(const vector<vector<int>>& adj, int src){
    int n=adj.size(); vector<int> dist(n,-1); queue<int> q;
    dist[src]=0; q.push(src);
    while(!q.empty()){ int u=q.front();q.pop();
        for(int v:adj[u]) if(dist[v]-1== -1-1){ dist[v]=dist[u]+1;q.push(v);} }
    return dist; }

// 2) BFS with parent reconstruction
pair<vector<int>,vector<int>> bfs_parent(const vector<vector<int>>& adj, int src){
    int n=adj.size(); vector<int> dist(n,-1),par(n,-1); queue<int> q;
    dist[src]=0; q.push(src);
    while(!q.empty()){ int u=q.front();q.pop();
        for(int v:adj[u]) if(dist[v]-1== -1-1){dist[v]=dist[u]+1;par[v]=u;q.push(v);}}
    return {dist,par}; }

// 3) Multi-source BFS on grid
vector<vector<int>> multisource_grid(const vector<string>& grid){
    int n=grid.size(),m=grid[0].size();
    vector<vector<int>> dist(n,vector<int>(m,-1));
    queue<pair<int,int>> q;
    for(int i=0;i<n;i++) for(int j=0;j<m;j++) if(grid[i][j]=='1'){dist[i][j]=0;q.push({i,j});}
    int di[]={0,0,-1,1},dj[]={-1,1,0,0};
    while(!q.empty()){ auto [i,j]=q.front();q.pop();
        for(int k=0;k<4;k++){ int ni=i+di[k],nj=j+dj[k];
            if(ni<0||ni>=n||nj<0||nj>=m||dist[ni][nj]!=-1) continue;
            dist[ni][nj]=dist[i][j]+1;q.push({ni,nj});}}
    return dist; }

// 4) 0-1 BFS (deque)
vector<int> bfs01(const vector<vector<pair<int,int>>>& adj, int src){
    const int INF=1e9; int n=adj.size();
    vector<int> dist(n,INF); deque<int> q;
    dist[src]=0; q.push_front(src);
    while(!q.empty()){ int u=q.front();q.pop_front();
        for(auto [v,w]:adj[u]) if(dist[u]+w<dist[v]){
            dist[v]=dist[u]+w; w?q.push_back(v):q.push_front(v);}}
    return dist; }

// 5) Kahn topological sort
vector<int> kahn(const vector<vector<int>>& adj){
    int n=adj.size(); vector<int> indeg(n,0);
    for(int u=0;u<n;u++) for(int v:adj[u]) indeg[v]++;
    queue<int> q;
    for(int i=0;i<n;i++) if(!indeg[i]) q.push(i);
    vector<int> order;
    while(!q.empty()){ int u=q.front();q.pop();order.push_back(u);
        for(int v:adj[u]) if(!--indeg[v]) q.push(v);}
    if((int)order.size()!=n) return {};
    return order; }
